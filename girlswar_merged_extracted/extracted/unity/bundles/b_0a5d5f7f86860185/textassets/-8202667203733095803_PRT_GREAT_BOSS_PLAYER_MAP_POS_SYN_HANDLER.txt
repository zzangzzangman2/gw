local t={}
function t:OnHandle(e)
local e=assert(GlobalPB.decode("PRT_GREAT_BOSS_PLAYER_MAP_POS_SYN",e));
if(GameInit.GetDebugLogProto())then
;
;
end
if ModulesInit.GreatBossMgr.isEnterStatus then
if ModulesInit.GreatBossDataMgr:GetHeroData(e.playerPos.playerId)==nil then
ModulesInit.GreatBossDataMgr:AddHeroData(e.playerPos)
ModulesInit.GreatBossMapMgr:AddHero(e.playerPos)
end
ModulesInit.GreatBossMgr:MoveHeroToPoint(e.playerPos)
local t=e.playerPos.mapPos==1 and e.playerPos.mapPos==e.playerPos.toMapPos
local a=ModulesInit.GreatBossDataMgr:GetHeroMapPos(PlayerMgr.PlayerInfo.uid)
if t or a==e.playerPos.mapPos then
ModulesInit.GreatBossDataMgr:RemovePosPlayer(e.playerPos.playerId)
end
end
return e
end
return t
