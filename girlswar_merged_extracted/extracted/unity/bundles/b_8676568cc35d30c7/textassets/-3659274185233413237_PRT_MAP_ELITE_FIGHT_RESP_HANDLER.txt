local t={}
function t:OnHandle(e)
local e=assert(GlobalPB.decode("PRT_MAP_ELITE_FIGHT_RESP",e));
if(GameInit.GetDebugLogProto())then
;
;
end
ModulesInit.ProcedureNormalBattle.IsRespBattleSuc=true
if ModulesInit.AutoHelperMgr.isAuto then
EventSystem.SendEvent(CommonEventId.OnBack_MAP_ELITE_FIGHT_RESP,e)
return
end
if e.award then
GameEntry.UI:OpenUIForm(UIFormId.UI_EliteBattleVictory,{award=e.award,mapId=ModulesInit.ProcedureNormalBattle.MapId})
else
local e=ModulesInit.ExpeditionManager.FightClearInfo
GameEntry.UI:OpenUIForm(UIFormId.UI_BattleFail,e)
end
return e
end
return t
