-- 秘境寻宝 · 购买藏宝图（DTSecretTreasureBuy：每行对应今日第几次购买档位；消耗道具相同、数量按档递增）
local ActSecretTreasureMgr = ModulesInit.ActSecretTreasureMgr
local DTItemDBModel = require('DataNode/DataTable/Create/item/DTItemDBModel')
local DTCurrencyDBModel = require('DataNode/DataTable/Create/constant/DTCurrencyDBModel')

local buyQuantity = 1
local initBuyMax = 1
local buyMax = 1
local goodsDid = 0
local currencyId = 0
local nameColorGroup = {}
local sureCallBback

local function totalRewardCountForBatch(completedToday, batchSize)
    local list = ActSecretTreasureMgr:GetBuyTicketCfgList()
    if not list then
        return 0
    end
    local sumCount = 0
    for index = 1, batchSize do
        local tierIndex = completedToday + index
        local row = list[tierIndex]
        if not row or not row.item then
            break
        end
        sumCount = sumCount + (tonumber(row.item[2]) or 1)
    end
    return sumCount
end

local function refreshBuyMaxLimit()
    local remainByLimit = ActSecretTreasureMgr:GetBuyTicketRemainingTodayCount()
    local completedToday = ActSecretTreasureMgr.buyCount or 0
    if remainByLimit <= 0 then
        return 0
    end
    local maxAffordable = 0
    for candidateSize = 1, remainByLimit do
        local costDid, sumCost = ActSecretTreasureMgr:GetBuyTicketTieredCostSum(completedToday, candidateSize)
        if not costDid or not sumCost then
            break
        end
        if ModulesInit.BagManager:CheckItemEnough(costDid, sumCost, false) then
            maxAffordable = candidateSize
        else
            break
        end
    end
    if maxAffordable > 0 then
        return maxAffordable
    end
    return 1
end

function OnInit(transform, userData)
    nameColorGroup = {"white","green","blue","purple","orange","red"}
    but_closs.onClick:AddListener(function()
        GameTools.CloseUIForm(UIFormId.UI_ActSecretTreasure_Buy)
    end)

    btn_jian10.onClick:AddListener(function()
        OnQuantityChange(-10)
    end)

    btn_jian.onClick:AddListener(function()
        OnQuantityChange(-1)
    end)

    btn_jia10.onClick:AddListener(function()
        OnQuantityChange(10)
    end)

    btn_jia.onClick:AddListener(function()
        OnQuantityChange(1)
    end)

    btn_max.onClick:AddListener(function()
        buyQuantity = buyMax
        OnQuantityChange(0)
    end)

    btn_return.onClick:AddListener(function()
        GameTools.CloseUIForm(UIFormId.UI_ActSecretTreasure_Buy)
    end)

    btn_confirm.onClick:AddListener(function()
        if buyMax <= 0 or buyQuantity <= 0 then
            return
        end
        if sureCallBback then
            sureCallBback(buyQuantity)
        end
        GameTools.CloseUIForm(UIFormId.UI_ActSecretTreasure_Buy)
    end)
end

function OnOpen(userData)
    sureCallBback = userData and userData.callBack
    buyQuantity = 1
    local list = ActSecretTreasureMgr:GetBuyTicketCfgList()
    if list and #list > 0 and list[1].item then
        goodsDid = list[1].item[1]
    else
        goodsDid = 0
    end
    RefreshPageInfo()
end

function OnClose()
    buyQuantity = 1
    initBuyMax = 1
    buyMax = 1
    goodsDid = 0
    currencyId = 0
    sureCallBback = nil
end

function RefreshPageInfo()
    local list = ActSecretTreasureMgr:GetBuyTicketCfgList()
    if not list or #list == 0 then
        return
    end
    local maxTier = #list
    local doneCount = ActSecretTreasureMgr.buyCount or 0
    if text_num_Limittimes then
        local limitText = GameTools.GetLocalize("UI_ActSecretTreasure_buy_limit_times", LanguageCategory.LangCommon, doneCount, maxTier)
        LuaUtils.SetLabelText(text_num_Limittimes, limitText)
    end

    initBuyMax = refreshBuyMaxLimit()
    if ActSecretTreasureMgr:GetBuyTicketRemainingTodayCount() <= 0 then
        buyMax = 0
        buyQuantity = 0
    else
        buyMax = initBuyMax
        if buyQuantity > buyMax then
            buyQuantity = buyMax
        end
        if buyQuantity < 1 then
            buyQuantity = 1
        end
    end

    local firstRow = list[1]
    currencyId = firstRow.consume[1]

    local item = DTCurrencyDBModel.GetEntity(currencyId)
    if not item then
        item = DTItemDBModel.GetEntity(currencyId)
    end
    local icon = UIUtil.GetItemIconPath(item.ItemIconSmall) or UIUtil.GetItemIconPath(item.iconSmall)
    LuaUtils.SetImageSprite(im_zuanshi_origin, icon, true)

    local rewardTotal = totalRewardCountForBatch(doneCount, math.max(buyQuantity, 1))
    UIUtil.SetItemCell(itemCell90.transform, { itemDid = goodsDid, count = rewardTotal })

    local nameStr, descStr = GetItemNameAndDesc(goodsDid)
    LuaUtils.SetTextMeshText(text_name, nameStr)
    LuaUtils.SetLabelText(text_reset, descStr)

    local ownedGoods = ModulesInit.BagManager:GetItemCountById(goodsDid)
    local ownedText = GameTools.GetLocalize("UI.Bag.Main.18", LanguageCategory.LangCommon, ownedGoods)
    LuaUtils.SetLabelText(text_cishu, ownedText)

    OnQuantityChange(0)
end

function OnQuantityChange(delta)
    local completedToday = ActSecretTreasureMgr.buyCount or 0
    initBuyMax = refreshBuyMaxLimit()
    if ActSecretTreasureMgr:GetBuyTicketRemainingTodayCount() <= 0 then
        buyMax = 0
        buyQuantity = 0
        OnButtonShow(completedToday)
        return
    end
    buyMax = initBuyMax

    buyQuantity = buyQuantity + delta
    if delta ~= 0 then
        if buyQuantity <= 0 then
            buyQuantity = 1
        end
        if buyQuantity >= buyMax then
            buyQuantity = buyMax
        end
    end

    local costDid, sumCost = ActSecretTreasureMgr:GetBuyTicketTieredCostSum(completedToday, buyQuantity)
    currencyId = costDid or currencyId

    local rewardTotal = totalRewardCountForBatch(completedToday, buyQuantity)
    UIUtil.SetItemCell(itemCell90.transform, { itemDid = goodsDid, count = rewardTotal })

    LuaUtils.SetLabelText(text_consumption, buyQuantity .. "/" .. buyMax)

    if sumCost then
        LuaUtils.SetLabelText(text_num_origin, tostring(sumCost))
    end

    OnButtonShow(completedToday)

    local contentFitter = trans_message.gameObject:GetComponent(typeof(CS.UnityEngine.UI.ContentSizeFitter))
    contentFitter:SetLayoutVertical()
    LuaUtils.RebuildLayout(trans_message.transform)
end

function OnButtonShow(completedToday)
    completedToday = completedToday or (ActSecretTreasureMgr.buyCount or 0)
    local costDid, sumCost = ActSecretTreasureMgr:GetBuyTicketTieredCostSum(completedToday, buyQuantity)
    local enough = false
    if costDid and sumCost and buyQuantity > 0 and buyMax > 0 then
        enough = ModulesInit.BagManager:CheckItemEnough(costDid, sumCost, false)
    end

    if enough then
        UIUtil.SetGrayByFont(btn_confirm.transform, false, false, GameInit.GrayFont2)
    else
        UIUtil.SetGrayByFont(btn_confirm.transform, true, false, GameInit.GrayFont2)
    end

    if buyQuantity <= 1 or buyMax <= 0 then
        UIUtil.SetGrayByFont(btn_jian.transform, true, false, GameInit.GrayFont2)
        UIUtil.SetGrayByFont(btn_jian10.transform, true, false, GameInit.GrayFont2)
    else
        UIUtil.SetGrayByFont(btn_jian.transform, false, false, GameInit.GrayFont2)
        UIUtil.SetGrayByFont(btn_jian10.transform, false, false, GameInit.GrayFont2)
    end

    if buyQuantity >= buyMax or buyMax <= 0 then
        UIUtil.SetGrayByFont(btn_jia.transform, true, false, GameInit.GrayFont2)
        UIUtil.SetGrayByFont(btn_jia10.transform, true, false, GameInit.GrayFont2)
        UIUtil.SetGrayByFont(btn_max.transform, true, false, GameInit.GrayFont2)
    else
        UIUtil.SetGrayByFont(btn_jia.transform, false, false, GameInit.GrayFont2)
        UIUtil.SetGrayByFont(btn_jia10.transform, false, false, GameInit.GrayFont2)
        UIUtil.SetGrayByFont(btn_max.transform, false, false, GameInit.GrayFont2)
    end

    if buyQuantity <= 1 and buyQuantity >= buyMax and buyMax > 0 then
        UIUtil.SetGrayByFont(bg_num.transform, true, false, GameInit.GrayFont2)
    else
        UIUtil.SetGrayByFont(bg_num.transform, false, false, GameInit.GrayFont2)
    end
end

function GetItemNameAndDesc(itemDid)
    local cfgInfo = ModulesInit.BagManager:GetBaseInfo(itemDid, true)
    text_name.color = DefaultColor[nameColorGroup[cfgInfo.color]]
    return cfgInfo.name, cfgInfo.desc
end

function OnWillClose()
    ViewMgr:OnWillClose(self.UIFormId)
end
