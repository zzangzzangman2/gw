local e=assert(GlobalPB.decode("PRT_FIGHT_MANUAL_RESP",e))
if(GameInit.GetDebugLogProto())then


end

if e==false then
local e=ModulesInit.ExpeditionManager.FightClearInfo
GameEntry.UI:OpenUIForm(UIFormId.UI_BattleFail,e)
end
elseif(ModulesInit.ProcedureNormalBattle.BattleType==BattleType.trial)then
if ModulesInit.AutoHelperMgr.isAuto then
EventSystem.SendEvent(CommonEventId.OnBack_FIGHT_MANUAL_RESP,e)
return
end
local a=ModulesInit.ExpeditionManager.FightClearState
local t=ModulesInit.ExpeditionManager.FightClearInfo
if a and e.award then

else
GameEntry.UI:OpenUIForm(UIFormId.UI_BattleFail,t)
end
