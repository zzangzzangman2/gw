local t=require('DataNode/DataTable/Create/maze/DTMazeItemShopEntity')
local e={}
local i=e
local a={};
local o={};
function e.LoadList()
local e
if(GameInit.ServerLoadTable)then
e=i.DeserializeFromJson()
else
e=t.GetArr()
end
for i=1,e.Len do
local e=e.ByIdx[i];
local e=t.NewFromArrItem(e)
a[#a+1]=e
o[e.id]=e
end
end
function e.GetList()
return a
end
function e.GetEntity(e)
return o[e]
end
function e.SerializeToJson()
local e=t.GetArr()
local e=JsonUtil.encode(e)
GameInit.WriteFile("DTMazeItemShopEntityData.txt",e)
end
function e.DeserializeFromJson()
local e=GameInit.ReadJsonData("DTMazeItemShopEntityData.txt")
return e
end
return e
