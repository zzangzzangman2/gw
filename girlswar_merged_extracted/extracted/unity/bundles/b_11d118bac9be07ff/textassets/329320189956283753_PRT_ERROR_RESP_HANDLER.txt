local t={}
function t:OnHandle(e)
local e=assert(GlobalPB.decode("PRT_ERROR_RESP",e));
if(GameInit.GetDebugLogProto())then
;
;
end
if e.errorCode==PROTO_ENUM.ErrCode.SERVER_ERROR then
EventSystem.SendEvent(CommonEventId.OnEventRespError,e)
else
UIUtil.ShowCommonTipsForCode(e.errorCode)
NetManager.CheckErrorColseForbid(e.protoCode)
EventSystem.SendEvent(CommonEventId.OnEventRespError,e)
end
end
return t
